@auth_router.post("/Criar_Conta")
async def criar_conta(Schema_Usuario: SchemaUsuario, session:Session = Depends(pegar_sessao)):
    # Verifica se já existe usuário com o mesmo email
    usuario = session.query(Usuario).filter(Usuario.email == Schema_Usuario.email).first()
    if usuario:
        raise HTTPException(status_code=400, detail="Usuário já existe")
    else:   
        # Criptografa a senha
        senha_criptada = bcript_context.hash(Schema_Usuario.senha)
        # Cria novo usuário (ajustando ordem dos parâmetros)
        novo_usuario = Usuario(Schema_Usuario.nome, Schema_Usuario.email, senha_criptada, Schema_Usuario.ativo, Schema_Usuario.admin)
        session.add(novo_usuario)
        session.commit()
        return {"mensagem": f"Conta criada com sucesso: {Schema_Usuario.email}"}

   - Defines a POST route at /auth/Criar_Conta.

   - session: SQLAlchemy Session object.
       - Obtained via Depends(pegar_sessao), which injects a database session into the route.

3. "usuario = session.query(Usuario).filter(Usuario.email == Schema_Usuario.email).first()"
   - Queries the Usuario table for a record with the same email as provided.

   - usuario: variable holding the result.

   - Checks if a user with the given email already exists.

---

6. "else:"
   - Executes if no user with the given email exists.
   - Proceeds to create a new account.

       - nome: user’s name.

       - ativo: active status.
       - admin: admin status.

9. "session.add(novo_usuario)"
   - Adds the new user object to the database session.
   - Prepares it for insertion.

10. "session.commit()"
    - Commits the transaction.
    - Saves the new user permanently in the database.

---

11. "return {'mensagem': f'Conta criada com sucesso: {Schema_Usuario.email}'}"
    - Returns a JSON response confirming account creation.
